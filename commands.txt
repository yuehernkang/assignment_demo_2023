CREATE KEYSPACE Messages
    WITH replication = {'class': 'NetworkTopologyStrategy'}
    AND durable_writes = true;

CREATE TABLE messages.Messages (
    chat_id TEXT,
    time TIMESTAMP,
    message TEXT,
    author TEXT,
    PRIMARY KEY(chat_id, time)
) WITH CLUSTERING ORDER BY (time DESC);
